25def compute_l2_pos_and_vel( 

26 time0: astropy.time.Time, earth_l2_distance_km: float = c.EARTH_L2_DISTANCE_KM 

27): 

28 """ 

29 Compute the position and velocity of the L2 Sun-Earth point at a given time. 

30 

31 The L2 point is not calculated using Lagrange's equations; instead, its 

32 distance from the Earth must be provided as the parameter `earth_l2_distance_km`. 

33 The default value is a reasonable estimate. The L2 point is assumed to lie along 

34 the line that connects the Solar System Barycenter with the Earth's gravitational 

35 center. 

36 

37 The return value is a 2-tuple containing two NumPy arrays: 

38 

39 1. A 3D array containing the XYZ components of the vector specifying the position 

40 of the L2 point, in km 

41 2. A 3D array containing the XYZ components of the velocity vector of the L2 point, 

42 in km/s 

43 

44 The two vectors are always roughly perpendicular, but they are not exactly due to 

45 the irregular motion of the Earth (caused by gravitational interactions with other 

46 Solar System bodies, like the Moon and Jupiter). 

47 """ 

48 # Use 

49 # 

50 # solar_system_ephemeris.set("builtin") 

51 # 

52 # to make AstroPy use the built-in ephemeris table 

53 

54 earth_pos, earth_vel = get_body_barycentric_posvel("earth", time0) 

55 earth_pos = ICRS(earth_pos).transform_to(DEFAULT_COORDINATE_SYSTEM).cartesian 

56 earth_vel = ICRS(earth_vel).transform_to(DEFAULT_COORDINATE_SYSTEM).cartesian 

57 

58 fudge_factor = earth_l2_distance_km / earth_pos.norm().to("km").value 

59 

60 # Move from Earth to L2 

61 l2_pos = earth_pos * (1.0 + fudge_factor) 

62 l2_vel = earth_vel * (1.0 + fudge_factor) 

63 

64 return ( 

65 l2_pos.xyz.to("km").value.transpose(), 

66 l2_vel.xyz.to("km/s").value.transpose(), 

67 )

70@njit 

71def compute_lissajous_pos_and_vel( 

72 time0, 

73 earth_angle_rad, 

74 earth_ang_speed_rad_s, 

75 radius1_km, 

76 radius2_km, 

77 ang_speed1_rad_s, 

78 ang_speed2_rad_s, 

79 phase_rad, 

80): 

81 """Compute the position and velocity of the spacecraft assuming a Lissajous orbit 

82 

83 The position and velocity are calculated in a reference frame centered on the L2 

84 point, whose axes are aligned with the Solar System Barycenter. This means that 

85 the position and velocity of the spacecraft with respect to the Solar System 

86 Barycenter itself can be calculated by summing the result of this function with 

87 the result of a call to :func:`.compute_l2_pos_and_vel`. 

88 """ 

89 

90 φ1 = ang_speed1_rad_s * time0 

91 φ2 = ang_speed2_rad_s * time0 

92 φ_earth = earth_ang_speed_rad_s * time0 

93 

94 pos_x, pos_y, pos_z = ( 

95 -radius1_km * np.sin(earth_angle_rad) * np.cos(φ1), 

96 radius1_km * np.cos(earth_angle_rad) * np.cos(φ1), 

97 radius2_km * np.sin(φ2 + phase_rad), 

98 ) 

99 

100 # This is the analytical derivative of the position (see above) 

101 cos1, sin1 = np.cos(φ1), np.sin(φ1) 

102 cos_earth, sin_earth = np.cos(φ_earth), np.sin(φ_earth) 

103 vel_x, vel_y, vel_z = ( 

104 -radius1_km 

105 * ( 

106 earth_ang_speed_rad_s * cos1 * cos_earth 

107 - ang_speed1_rad_s * sin1 * sin_earth 

108 ), 

109 -radius1_km 

110 * ( 

111 earth_ang_speed_rad_s * cos1 * sin_earth 

112 + ang_speed1_rad_s * sin1 * cos_earth 

113 ), 

114 ang_speed2_rad_s * radius2_km * np.cos(φ2 + phase_rad), 

115 ) 

116 

117 return pos_x, pos_y, pos_z, vel_x, vel_y, vel_z

120@njit 

121def sum_lissajous_pos_and_vel( 

122 pos, 

123 vel, 

124 start_time_s, 

125 end_time_s, 

126 radius1_km, 

127 radius2_km, 

128 ang_speed1_rad_s, 

129 ang_speed2_rad_s, 

130 phase_rad, 

131): 

132 """Add the position and velocity of a Lissajous orbit to some positions/velocities 

133 

134 The `pos` and `vel` arrays must have the same shape (N×3, with N being the number 

135 of position-velocity pairs); the 3D position and velocity of the Lissajous orbit 

136 will be added to these arrays. 

137 

138 The `times` array must be an array of *float* values, typically starting from zero, 

139 and they must measure the time in seconds. All the other parameters are usually 

140 taken from a :class:`.SpacecraftOrbit` object. 

141 

142 This function has no return value, as the result is stored in `pos` and `vel`.""" 

143 

144 # This code might look weird if you have never used Numba. However, it is extremely 

145 # efficient, as it completely avoids memory allocations. 

146 

147 n_elements = pos.shape[ 

148 0 

149 ] # With Numba 0.52, you cannot use len(…) on a NumPy array yet 

150 time_s = start_time_s 

151 delta_time_s = (end_time_s - start_time_s) / n_elements 

152 

153 for idx in range(n_elements): 

154 l2_pos_x, l2_pos_y, l2_pos_z = pos[idx, :] 

155 l2_vel_x, l2_vel_y, l2_vel_z = vel[idx, :] 

156 

157 # The parameter `earth_angle_rad` is the angular position of the Earth (in 

158 # radians) on the Ecliptic plane with respect to the Vernal Equinox. If we 

159 # denote with r = (x, y) the projected position of the L2 point on the 

160 # Ecliptic plane, then the angle is just atan(y/x). 

161 earth_angle_rad = np.arctan2(l2_pos_y, l2_pos_x) 

162 

163 # `earth_ang_speed_rad_s` is the angular speed of the Earth on the Ecliptic 

164 # plane (in rad/s). We calculate it by projecting the motion of L2 on the 

165 # Ecliptic plane and determining the tangential component of its velocity. 

166 # The vector perpendicular to it and directed counterclockwise is 

167 # 

168 # | 0 −1 | | x | | −y | 

169 # e⟂ = | | · | | = | |, 

170 # | 1 0 | | y | | x | 

171 # 

172 # which must be normalized, so that e⟂ = (−y, x) / √(x² + y²). Then, the 

173 # component of the velocity vector aligned along this vector is v·e⟂. 

174 # However, we are interested in the *angular* speed, so we must use the 

175 # relation v·e⟂ = ω r, which leads to 

176 # 

177 # ω = v·e⟂ / r 

178 # = (v_x, v_y) · (−y / √(x² + y²), x / √(x² + y²)) / √(x² + y²) = 

179 # = (−v_x · y + v_y · x) / (x² + y²) 

180 radius_squared = l2_pos_x**2 + l2_pos_y**2 

181 earth_ang_speed_rad_s = ( 

182 -l2_vel_x * l2_pos_y + l2_vel_y * l2_pos_x 

183 ) / radius_squared 

184 

185 posx, posy, posz, velx, vely, velz = compute_lissajous_pos_and_vel( 

186 time0=time_s, 

187 earth_angle_rad=earth_angle_rad, 

188 earth_ang_speed_rad_s=earth_ang_speed_rad_s, 

189 radius1_km=radius1_km, 

190 radius2_km=radius2_km, 

191 ang_speed1_rad_s=ang_speed1_rad_s, 

192 ang_speed2_rad_s=ang_speed2_rad_s, 

193 phase_rad=phase_rad, 

194 ) 

195 pos[idx][0] += posx 

196 pos[idx][1] += posy 

197 pos[idx][2] += posz 

198 

199 vel[idx][0] += velx 

200 vel[idx][1] += vely 

201 vel[idx][2] += velz 

202 

203 time_s += delta_time_s

206@dataclass 

207class SpacecraftOrbit: 

208 """A dataclass describing the orbit of the LiteBIRD spacecraft 

209 

210 This structure has the following fields: 

211 

212 - `start_time`: Date and time when the spacecraft starts its nominal orbit 

213 - `earth_l2_distance_km`: distance between the Earth's barycenter and the L2 

214 point, in km 

215 - `radius1_km`: first radius describing the Lissajous orbit followed by the 

216 spacecraft, in km 

217 - `radius2_km`: second radius describing the Lissajous orbit followed by the 

218 spacecraft, in km 

219 - `ang_speed1_rad_s`: first angular speed of the Lissajous orbit, in rad/s 

220 - `ang_speed2_rad_s`: second angular speed of the Lissajous orbit, in rad/s 

221 - `phase_rad`: phase difference between the two periodic motions in the Lissajous 

222 orbit, in radians 

223 - `solar_velocity_km_s`: velocity of the Sun as estimated from Planck 2018 Solar 

224 dipole (see arxiv: 1807.06207) 

225 - `solar_velocity_gal_lat_rad`: galactic latitude direction of the Planck 2018 

226 Solar dipole 

227 - `solar_velocity_gal_lon_rad`: galactic longitude direction of the Planck 2018 

228 Solar dipole 

229 

230 The default values are the nominal numbers of the orbit followed by WMAP, described 

231 in Cavaluzzi, Fink & Coyle (2008). 

232 """ 

233 

234 start_time: astropy.time.Time 

235 earth_l2_distance_km: float = c.EARTH_L2_DISTANCE_KM 

236 radius1_km: float = 244_450.0 

237 radius2_km: float = 137_388.0 

238 ang_speed1_rad_s: float = cycles_per_year_to_rad_per_s(2.02104) 

239 ang_speed2_rad_s: float = cycles_per_year_to_rad_per_s(1.98507) 

240 phase_rad: float = np.deg2rad(-47.944) 

241 solar_velocity_km_s: float = c.SOLAR_VELOCITY_KM_S 

242 solar_velocity_gal_lat_rad: float = c.SOLAR_VELOCITY_GAL_LAT_RAD 

243 solar_velocity_gal_lon_rad: float = c.SOLAR_VELOCITY_GAL_LON_RAD 

244 

245 solar_velocity_ecl_xyz_km_s = ( 

246 SkyCoord( 

247 solar_velocity_gal_lon_rad, 

248 solar_velocity_gal_lat_rad, 

249 unit="rad", 

250 frame="galactic", 

251 ) 

252 .transform_to(DEFAULT_COORDINATE_SYSTEM) 

253 .cartesian.get_xyz() 

254 .value 

255 * solar_velocity_km_s 

256 )

259class SpacecraftPositionAndVelocity: 

260 """Encode the position/velocity of the spacecraft with respect to the Solar System 

261 

262 This class contains information that characterize the motion of the spacecraft. 

263 It is mainly useful to simulate the so-called CMB «orbital dipole» and to properly 

264 check the visibility of the Sun, the Moon and inner planets. The coordinate 

265 system used by this class is the standard Barycentric Ecliptic reference frame. 

266 

267 The fields of this class are the following: 

268 

269 - ``orbit``: a :class:`.SpacecraftOrbit` object used to compute the positions and 

270 velocities in this object; 

271 

272 - ``start_time``: the time when the nominal orbit started (a ``astropy.time.Time`` 

273 object); 

274 

275 - ``time_span_s``: the time span covered by this object, in seconds; 

276 

277 - ``positions_km``: a ``N×3`` matrix, representing a list of ``N`` XYZ vectors 

278 encoding the position of the spacecraft in the Barycentric 

279 Ecliptic reference frame (in kilometers); 

280 

281 - ``velocities_km_s``: a ``N×3`` matrix, representing a list of ``N`` XYZ vectors 

282 encoding the linear velocity of the spacecraft in the 

283 Barycentric Ecliptic reference frame (in km/s). 

284 

285 """ 

286

288 self, 

289 orbit: SpacecraftOrbit,

291 time_span_s: float,

309 def compute_velocities( 

310 self, time0: astropy.time.Time, delta_time_s: float, num_of_samples: int 

311 ): 

312 """Perform a linear interpolation to sample the satellite velocity in time 

313 

314 Return a N×3 array containing a set of `num_of_samples` 3D vectors with the 

315 velocity of the spacecraft (in km/s) computed every `delta_time_s` seconds 

316 starting from time `time0`. 

317 """ 

318 delta_start_s = (time0 - self.start_time).sec 

319 t = delta_start_s + np.linspace( 

320 start=0.0, 

321 stop=delta_time_s * num_of_samples, 

322 endpoint=False, 

323 num=num_of_samples, 

324 ) 

325 tp = np.linspace( 

326 start=0.0, stop=self.time_span_s, num=self.velocities_km_s.shape[0] 

327 ) 

328 

329 velocities = np.empty((num_of_samples, 3)) 

330 velocities[:, 0] = np.interp(x=t, xp=tp, fp=self.velocities_km_s[:, 0]) 

331 velocities[:, 1] = np.interp(x=t, xp=tp, fp=self.velocities_km_s[:, 1]) 

332 velocities[:, 2] = np.interp(x=t, xp=tp, fp=self.velocities_km_s[:, 2]) 

333 

334 return velocities

337def compute_start_and_span_for_obs( 

338 observations: Observation | list[Observation], 

339) -> tuple[astropy.time.Time, float]: 

340 """ 

341 Compute the start time and the overall duration in seconds of a set of observations. 

342 

343 The code returns the earliest start time of the observations in `observations` as 

344 well as their overall time span. Gaps between observations are neglected. 

345 """ 

346 

347 if isinstance(observations, Observation): 

348 obs_list = [observations] 

349 else: 

350 obs_list = observations 

351 

352 start_time, end_time = None, None 

353 for cur_obs in obs_list: 

354 assert isinstance(cur_obs.start_time, astropy.time.Time), ( 

355 "You must use astropy.time.Time in Observation objects" 

356 ) 

357 

358 cur_start_time = cur_obs.start_time 

359 if (start_time is None) or (cur_start_time < start_time): 

360 start_time = cur_start_time 

361 

362 cur_end_time = cur_obs.start_time + cur_obs.get_time_span() 

363 if (end_time is None) or (end_time > cur_end_time): 

364 end_time = cur_end_time 

365 

366 time_span_s = (end_time - start_time).to("s").value 

367 

368 return start_time, time_span_s

371def spacecraft_pos_and_vel( 

372 orbit: SpacecraftOrbit, 

373 observations: Observation | list[Observation] | None = None, 

374 start_time: astropy.time.Time | None = None, 

375 time_span_s: float | None = None, 

376 delta_time_s: float = 86400.0, 

377) -> SpacecraftPositionAndVelocity: 

378 """Compute the position and velocity of the L2 point within some time span 

379 

380 This function computes the XYZ position and velocity of the second Sun-Earth 

381 Lagrangean point (L2) over a time span specified either by a 

382 :class:`.Observation` object/list of objects, or by an explicit pair of values 

383 `start_time` (an ``astropy.time.Time`` object) and `time_span_s` (length in 

384 seconds). The position is specified in the standard Barycentric Ecliptic 

385 reference frame. 

386 

387 The position of the L2 point is computed starting from the position of the 

388 Earth and moving away along the anti-Sun direction by a number of kilometers 

389 equal to `earth_l2_distance_km`. 

390 

391 The result is an object of type :class:`.SpacecraftPositionAndVelocity`. 

392 

393 If SpacecraftOrbit.solar_velocity_km_s > 0 also the Sun velocity in the rest 

394 frame of the CMB is added to the total velocity of the spacecraft. 

395 """ 

396 assert observations or (start_time and time_span_s), ( 

397 "You must either provide a Observation or start_time/time_span_s" 

398 ) 

399 

400 if observations: 

401 # The caller either provided an observation or a list of observations. 

402 # Let's compute the overall time span 

403 start_time, time_span_s = compute_start_and_span_for_obs(observations) 

404 

405 # We are going to compute the position of the L2 point at N times. The value N 

406 # is chosen such that the spacing between two consecutive times is never longer 

407 # than ~1 day 

408 times = astropy.time.TimeDelta( 

409 np.linspace( 

410 start=0.0, 

411 stop=time_span_s, 

412 num=int(np.ceil(time_span_s / delta_time_s)) + 1, 

413 ), 

414 format="sec", 

415 scale=DEFAULT_TIME_SCALE, 

416 ) 

417 

418 pos, vel = compute_l2_pos_and_vel( 

419 time0=start_time + times, earth_l2_distance_km=orbit.earth_l2_distance_km 

420 ) 

421 

422 if orbit.radius1_km > 0 or orbit.radius2_km > 0: 

423 sum_lissajous_pos_and_vel( 

424 pos=pos, 

425 vel=vel, 

426 start_time_s=(start_time - orbit.start_time).to("s").value, 

427 end_time_s=time_span_s, 

428 radius1_km=orbit.radius1_km, 

429 radius2_km=orbit.radius2_km, 

430 ang_speed1_rad_s=orbit.ang_speed1_rad_s, 

431 ang_speed2_rad_s=orbit.ang_speed2_rad_s, 

432 phase_rad=orbit.phase_rad, 

433 ) 

434 

435 if orbit.solar_velocity_km_s > 0: 

436 vel += orbit.solar_velocity_ecl_xyz_km_s[np.newaxis, :] 

437 

438 return SpacecraftPositionAndVelocity( 

439 orbit=orbit, 

440 start_time=start_time, 

441 time_span_s=time_span_s, 

442 positions_km=pos, 

443 velocities_km_s=vel, 

444 )
