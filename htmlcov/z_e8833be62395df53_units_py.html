94 if unit == Units.ADU:

100 @staticmethod 

101 def get_conversion_factor( 

102 unit_from: Units, unit_to: Units, freq_ghz: float | None = None 

103 ) -> float: 

104 """ 

105 Calculates the multiplicative factor to convert between units using Astropy equivalencies. 

106 

107 Formula: value_out = value_in * factor 

108 

109 Parameters 

110 ---------- 

111 unit_from : Units 

112 The source unit. 

113 unit_to : Units 

114 The destination unit. 

115 freq_ghz : float, optional 

116 Frequency in Gigahertz (GHz). Strictly required for conversions involving 

117 Flux (MJy/sr) or transformations between CMB and RJ temperatures. 

118 

119 Returns 

120 ------- 

121 float 

122 The scalar conversion factor. 

123 

124 Raises 

125 ------ 

126 ValueError 

127 If ADU is involved (requires instrument gain) or if frequency is missing 

128 for spectral conversions. 

129 """ 

130 if unit_from == unit_to: 

131 return 1.0 

132 

133 if Units.ADU in [unit_from, unit_to]: 

134 raise ValueError( 

135 "Conversion involving ADU requires instrument Gain and cannot be handled by simple unit conversion." 

136 ) 

137 

138 # Define CMB equivalencies (required for T_RJ <-> T_CMB <-> Flux) 

139 eq = [] 

140 if freq_ghz is not None: 

141 eq = u.cmb_equivalencies(freq_ghz * getattr(u, "GHz")) 

142 

143 # We rely on Astropy to handle the physics. 

144 # We create a Quantity of "1.0 unit_from" and convert it to "unit_to". 

145 try: 

146 inp_quantity = 1.0 * UnitUtils.get_astropy_unit(unit_from) 

147 out_value = inp_quantity.to_value( 

148 UnitUtils.get_astropy_unit(unit_to), equivalencies=eq 

149 ) 

150 return out_value 

151 

152 except u.UnitConversionError as e: 

153 # Raise a more descriptive error if the frequency is missing for spectral conversions 

154 if freq_ghz is None and ( 

155 UnitUtils.is_flux(unit_from) 

156 or UnitUtils.is_flux(unit_to) 

157 or "CMB" in unit_from.value 

158 or "CMB" in unit_to.value 

159 ): 

160 raise ValueError( 

161 f"Frequency (freq_ghz) is required to convert between {unit_from.name} and {unit_to.name}" 

162 ) from e 

163 raise e
