70@njit 

71def _solve_binning(nobs_matrix, atd): 

72 # Solve the map-making equation 

73 # 

74 # This method alters the parameter `nobs_matrix`, so that after its completion 

75 # each 3×3 matrix in nobs_matrix[idx, :, :] will be the *inverse*. 

76 

77 # Expected shape: 

78 # - `nobs_matrix`: (N_p, 3, 3) is an array of N_p 3×3 matrices, where 

79 # N_p is the number of pixels in the map 

80 # - `atd`: (N_p, 3) 

81 npix = atd.shape[0] 

82 

83 for ipix in range(npix): 

84 if np.linalg.cond(nobs_matrix[ipix]) < COND_THRESHOLD: 

85 atd[ipix] = np.linalg.solve(nobs_matrix[ipix], atd[ipix]) 

86 nobs_matrix[ipix] = np.linalg.inv(nobs_matrix[ipix]) 

87 else: 

88 nobs_matrix[ipix].fill(hp.UNSEEN) 

89 atd[ipix].fill(hp.UNSEEN)

92@njit 

93def _accumulate_samples_and_build_nobs_matrix( 

94 tod: npt.NDArray, 

95 pix: npt.NDArray, 

96 psi: npt.NDArray, 

97 weights: npt.NDArray, 

98 d_mask: npt.NDArray, 

99 t_mask: npt.NDArray, 

100 nobs_matrix: npt.NDArray, 

101 *, 

102 additional_component: bool, 

103) -> None: 

104 # Fill the upper triangle of the N_obs matrix and use the lower 

105 # triangle for the RHS of the map-making equation: 

106 # 

107 # 1. The upper triangle and the diagonal contains the coefficients in 

108 # Eq. (10) of KurkiSuonio2009. This must be set just once, as it only 

109 # depends on the pointing information. The flag `additional_component` 

110 # tells if this part must be calculated (``False``) or not 

111 # 2. The lower triangle contains the weighted sum of I/Q/U, i.e., 

112 # 

113 # (I + Q·cos(2ψ) + U·sin(2ψ)) / σ² 

114 

115 assert tod.shape == pix.shape == psi.shape 

116 

117 assert tod.shape[0] == d_mask.shape[0] 

118 

119 num_of_detectors = tod.shape[0] 

120 

121 for idet in range(num_of_detectors): 

122 if not d_mask[idet]: 

123 continue 

124 

125 inv_sigma = 1.0 / np.sqrt(weights[idet]) 

126 inv_sigma2 = inv_sigma * inv_sigma 

127 

128 if not additional_component: 

129 # Fill the upper triangle 

130 for cur_pix_idx, cur_psi, cur_t_mask in zip(pix[idet], psi[idet], t_mask): 

131 if cur_t_mask: 

132 cos_over_sigma = np.cos(2 * cur_psi) * inv_sigma 

133 sin_over_sigma = np.sin(2 * cur_psi) * inv_sigma 

134 info_pix = nobs_matrix[cur_pix_idx] 

135 

136 # Upper triangle 

137 info_pix[0, 0] += inv_sigma2 

138 info_pix[0, 1] += inv_sigma * cos_over_sigma 

139 info_pix[0, 2] += inv_sigma * sin_over_sigma 

140 info_pix[1, 1] += cos_over_sigma * cos_over_sigma 

141 info_pix[1, 2] += sin_over_sigma * cos_over_sigma 

142 info_pix[2, 2] += sin_over_sigma * sin_over_sigma 

143 

144 # Fill the lower triangle 

145 for cur_sample, cur_pix_idx, cur_psi, cur_t_mask in zip( 

146 tod[idet, :], pix[idet, :], psi[idet, :], t_mask 

147 ): 

148 if cur_t_mask: 

149 cos_over_sigma = np.cos(2 * cur_psi) * inv_sigma 

150 sin_over_sigma = np.sin(2 * cur_psi) * inv_sigma 

151 info_pix = nobs_matrix[cur_pix_idx] 

152 

153 info_pix[1, 0] += cur_sample * inv_sigma2 

154 info_pix[2, 0] += cur_sample * cos_over_sigma * inv_sigma 

155 info_pix[2, 1] += cur_sample * sin_over_sigma * inv_sigma

158@njit 

159def _numba_extract_map_and_fill_nobs_matrix( 

160 nobs_matrix: npt.NDArray, rhs: npt.NDArray 

161) -> None: 

162 # This is used internally by _extract_map_and_fill_info. The function 

163 # modifies both `info` and `rhs`; the first parameter would be an `inout` 

164 # parameter in Fortran (it is both used as input and output), while `rhs` 

165 # is an `out` parameter 

166 for idx in range(nobs_matrix.shape[0]): 

167 # Extract the vector from the lower left triangle of the 3×3 matrix 

168 # nobs_matrix[idx, :, :] 

169 rhs[idx, 0] = nobs_matrix[idx, 1, 0] 

170 rhs[idx, 1] = nobs_matrix[idx, 2, 0] 

171 rhs[idx, 2] = nobs_matrix[idx, 2, 1] 

172 

173 # Make each 3×3 matrix in nobs_matrix[idx, :, :] symmetric 

174 nobs_matrix[idx, 1, 0] = nobs_matrix[idx, 0, 1] 

175 nobs_matrix[idx, 2, 0] = nobs_matrix[idx, 0, 2] 

176 nobs_matrix[idx, 2, 1] = nobs_matrix[idx, 1, 2] 

177

179def _extract_map_and_fill_info(info: npt.NDArray) -> npt.NDArray: 

180 # Extract the RHS of the mapmaking equation from the lower triangle of info 

181 # and fill the lower triangle with the upper triangle, thus making each 

182 # matrix in "info" symmetric 

183 rhs = np.empty((info.shape[0], 3), dtype=info.dtype) 

184 

185 # The implementation in Numba of this code is ~5 times faster than the older 

186 # implementation that used NumPy. 

187 _numba_extract_map_and_fill_nobs_matrix(info, rhs) 

188 

189 return rhs

192def _build_nobs_matrix( 

193 nside: int, 

194 obs_list: list[Observation], 

195 ptg_list: list[npt.NDArray | Callable], 

196 hwp: HWP | None, 

197 dm_list: list[npt.NDArray], 

198 tm_list: list[npt.NDArray], 

199 output_coordinate_system: CoordinateSystem, 

200 components: list[str], 

201 pointings_dtype=np.float64, 

202) -> npt.NDArray: 

203 hpx = Healpix_Base(nside, "RING") 

204 n_pix = nside_to_npix(nside) 

205 

206 nobs_matrix = np.zeros((n_pix, 3, 3)) 

207 

208 for obs_idx, (cur_obs, cur_ptg, cur_d_mask, cur_t_mask) in enumerate( 

209 zip(obs_list, ptg_list, dm_list, tm_list) 

210 ): 

211 cur_weights = get_map_making_weights(cur_obs, check=True) 

212 

213 # Determine the HWP angle to use: 

214 # - If an external HWP object is provided, compute the angle from it 

215 # - If not, compute or retrieve the HWP angle from the observation, depending on availability 

216 hwp_angle = _get_hwp_angle(obs=cur_obs, hwp=hwp, pointing_dtype=pointings_dtype) 

217 

218 pixidx_all, polang_all = _compute_pixel_indices( 

219 hpx=hpx, 

220 pointings=cur_ptg, 

221 pol_angle_detectors=cur_obs.pol_angle_rad, 

222 num_of_detectors=cur_obs.n_detectors, 

223 num_of_samples=cur_obs.n_samples, 

224 hwp_angle=hwp_angle, 

225 output_coordinate_system=output_coordinate_system, 

226 pointings_dtype=pointings_dtype, 

227 ) 

228 

229 first_component = getattr(cur_obs, components[0]) 

230 for idx, cur_component_name in enumerate(components): 

231 cur_component = getattr(cur_obs, cur_component_name) 

232 assert cur_component.shape == first_component.shape, ( 

233 'The two TODs "{}" and "{}" do not have a matching shape'.format( 

234 components[0], cur_component_name 

235 ) 

236 ) 

237 _accumulate_samples_and_build_nobs_matrix( 

238 cur_component, 

239 pixidx_all, 

240 polang_all, 

241 cur_weights, 

242 cur_d_mask, 

243 cur_t_mask, 

244 nobs_matrix, 

245 additional_component=idx > 0, 

246 ) 

247 

248 del pixidx_all, polang_all 

249 

250 assert obs_list, "No observations provided" 

251 if mpi.MPI_ENABLED: 

252 from litebird_sim.mpi import MPI 

253 if all([obs.comm is None for obs in obs_list]) or not mpi.MPI_ENABLED: 

254 # Serial call 

255 pass 

256 elif all( 

257 [ 

258 MPI.Comm.Compare(obs_list[i].comm, obs_list[i + 1].comm) < 2 

259 for i in range(len(obs_list) - 1) 

260 ] 

261 ): 

262 obs_list[0].comm.Allreduce(MPI.IN_PLACE, nobs_matrix, MPI.SUM) 

263 

264 else: 

265 raise NotImplementedError( 

266 "All observations must be distributed over the same MPI groups" 

267 ) 

268 

269 return nobs_matrix

272def make_binned_map( 

273 nside: int, 

274 observations: Observation | list[Observation], 

275 pointings: np.ndarray | list[np.ndarray] | None = None, 

276 hwp: HWP | None = None, 

277 output_coordinate_system: CoordinateSystem = CoordinateSystem.Galactic, 

278 components: str | list[str] = "tod", 

279 detector_split: str = "full", 

280 time_split: str = "full", 

281 pointings_dtype=np.float64, 

282) -> BinnerResult: 

283 """Bin Map-maker 

284 

285 Map a list of observations 

286 

287 Args: 

288 observations (list of :class:`Observations`): observations to be mapped. 

289 They are required to have the following attributes as arrays 

290 

291 * `pointings`: the pointing information (in radians) for each tod 

292 sample. It must be a tensor with shape ``(N_d, N_t, 3)``, 

293 with ``N_d`` number of detectors and ``N_t`` number of 

294 samples in the TOD. 

295 * any attribute listed in `components` (by default, `tod`) and 

296 containing the TOD(s) to be binned together. 

297 

298 If the observations are distributed over some communicator(s), they 

299 must share the same group processes. 

300 If pointings and psi are not included in the observations, they can 

301 be provided through an array (or a list of arrays) of dimension 

302 (Ndetectors x Nsamples x 3), containing theta, phi and psi 

303 nside (int): HEALPix nside of the output map 

304 pointings (array or list of arrays): optional, external pointing 

305 information, if not included in the observations 

306 hwp (HWP, optional): An instance of the :class:`.HWP` class (optional) 

307 output_coordinate_system (:class:`.CoordinateSystem`): the coordinates 

308 to use for the output map 

309 components (list[str] or str): components to include in the map-making. 

310 The default is just to use the field ``tod`` of each 

311 :class:`.Observation` object 

312 detector_split (str): select the detector split to use in the map-making 

313 time_split (str): select the time split to use in the map-making. 

314 pointings_dtype(dtype): data type for pointings generated on the fly. If 

315 the pointing is passed or already precomputed this parameter is 

316 ineffective. Default is `np.float64`. 

317 

318 Returns: 

319 An instance of the class :class:`.MapMakerResult`. If the observations are 

320 distributed over MPI Processes, all of them get a copy of the same object. 

321 """ 

322 

323 if isinstance(components, str): 

324 components = [components] 

325 

326 obs_list, ptg_list = _normalize_observations_and_pointings( 

327 observations=observations, pointings=pointings 

328 ) 

329 assert ptg_list, "No observations provided" 

330 

331 detector_mask_list = _build_mask_detector_split(detector_split, obs_list) 

332 

333 time_mask_list = _build_mask_time_split(time_split, obs_list) 

334 

335 nobs_matrix = _build_nobs_matrix( 

336 nside=nside, 

337 obs_list=obs_list, 

338 ptg_list=ptg_list, 

339 hwp=hwp, 

340 dm_list=detector_mask_list, 

341 tm_list=time_mask_list, 

342 output_coordinate_system=output_coordinate_system, 

343 components=components, 

344 pointings_dtype=pointings_dtype, 

345 ) 

346 

347 rhs = _extract_map_and_fill_info(nobs_matrix) 

348 

349 _solve_binning(nobs_matrix, rhs) 

350 

351 return BinnerResult( 

352 binned_map=rhs.T, 

353 invnpp=nobs_matrix, 

354 coordinate_system=output_coordinate_system, 

355 components=components, 

356 detector_split=detector_split, 

357 time_split=time_split, 

358 )

361def check_valid_splits( 

362 observations: Observation | list[Observation], 

363 detector_splits: str | list[str] = "full", 

364 time_splits: str | list[str] = "full", 

365): 

366 """Check if the splits are valid 

367 

368 For each observation in the list, check if the detector and time splits 

369 are valid. 

370 In particular, the compatibility between the detectors in each observation 

371 and the desired split in detector domain is checked. On the other hand, this 

372 assess whether the desired time split fits inside the duration of the 

373 observation (when this applies). 

374 If the splits are not compatible with the input data, an error is raised. 

375 

376 Args: 

377 observations (list of :class:`Observations`): observations to be mapped. 

378 They are required to have the following attributes as arrays 

379 

380 * `pointings`: the pointing information (in radians) for each tod 

381 sample. It must be a tensor with shape ``(N_d, N_t, 3)``, 

382 with ``N_d`` number of detectors and ``N_t`` number of 

383 samples in the TOD. 

384 * any attribute listed in `components` (by default, `tod`) and 

385 containing the TOD(s) to be binned together. 

386 

387 If the observations are distributed over some communicator(s), they 

388 must share the same group processes. 

389 If pointings and psi are not included in the observations, they can 

390 be provided through an array (or a list of arrays) of dimension 

391 (Ndetectors x Nsamples x 3), containing theta, phi and psi 

392 detector_splits (str | list[str], optional): detector-domain splits 

393 used to produce maps. 

394 

395 * "full": every detector in the observation will be used; 

396 * "waferXXX": the mapmaking will be performed on the intersection 

397 of the detectors specified in the input and the detectors specified 

398 in the detector_split. 

399 The wafer must be specified in the format "waferXXX". The valid values 

400 for "XXX" are all the 3-digits strings corresponding to the wafers 

401 in the LITEBIRD focal plane (e.g. L00, M01, H02). 

402 

403 time_splits (str | list[str], optional): time-domain splits 

404 used to produce maps. This defaults to "full" indicating that every 

405 sample in the observation will be used. In addition, the user can specify 

406 a string, or a list of strings, to indicate a subsample of the observation 

407 to be used: 

408 

409 * "full": every sample in the observation will be used; 

410 * "first_half" and/or "second_half": the first and/or second half of the 

411 observation will be used; 

412 * "odd" and/or "even": the odd and/or even samples in the observation 

413 will be used; 

414 * "yearX": the samples in the observation will be 

415 used according to the year they belong to (relative to the 

416 starting time). The valid values for "X" are ["1", "2", "3"]. 

417 * "surveyX": the samples in the observation will be used according 

418 to the requested survey. In this context, a survey is taken to 

419 be complete in 6 months, thus the valid values for "X" are 

420 ["1", "2", "3", "4", "5", "6"]. 

421 

422 """
